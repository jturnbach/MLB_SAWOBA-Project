bips <- bips %>%

    stand == "R" & spray_angle < -20 ~ "pulled",
    stand == "R" & spray_angle >= -20 & spray_angle <= 20 ~ "center",

    stand == "L" & spray_angle < -20 ~ "opposite",
    stand == "L" & spray_angle >= -20 & spray_angle <= 20 ~ "center",
